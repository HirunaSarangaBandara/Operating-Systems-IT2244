/*
Q4) Get the days based on the user input number

Output:
		Enter the number: 5
		Friday

*/

touch day.sh
vi day.sh

echo "Enter the number: "
read x
case $x in
1)echo "Monday";;
2)echo "Tuesday";;
3)echo "Wednesday";;
4)echo "Thursday";;
5)echo "Friday";;
6)echo "Saturday";;
7)echo "Sunday";;
*)echo "Invalid Number";;
esac

chmod 777 day.sh
./day.sh

6

		1 9
		1 is less than 9

*/

touch max.sh
vi max.sh

echo "Enter two numbers:"
read num1
echo "Enter two numbers:"
read num2

if [ $num1 -gt $num2 ]; then
  echo "$num1 is greater than $num2"
elif [ $num1 -lt $num2 ]; then
  echo "$num1 is less than $num2"
else
